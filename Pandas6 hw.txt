import pandas as pd

def delete_duplicate_emails(person: pd.DataFrame) -> None:
    person.sort_values(by='id',ascending=True,inplace=True)
    person.drop_duplicates(['email'], inplace=True)

import pandas as pd

def delete_duplicate_emails(person: pd.DataFrame) -> None:
    min_id= person.groupby('email')['id'].transform('min')
    idtoberemoved = person[person['id'] != min_id]
    person.drop(idtoberemoved.index ,inplace=True)

import pandas as pd

def count_rich_customers(store: pd.DataFrame) -> pd.DataFrame:
    df = store[store['amount'] > 500]
    richcount =len(set(df['customer_id']))
    return pd.DataFrame([richcount],columns=['rich_count'])
